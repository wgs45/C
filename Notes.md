### Create new project in react js

# Change directory to destination directory

cd coding/project

# Then run the following command

npm create vite@latest

# Enter the project name based on your preference

project_name

# Choose library name

React

# Choose programming language

Javascript (Only JavaScript)

# Change directory to the new project directory

cd project_name

# Then run the following command to install required libraries

npm install

# Then run the following command to start the project

npm run dev

## Install the required libraries for the project 

# Install react router dom

npm i react-router-dom

# Install react icons

npm i react-icons

### Create new project in react Native

# change directory to destination directory

cd coding/react_native

# Create new project (Enter the project name based on your preference)

npx react-native init project_name

# Wait for the project to finish the installation

estimated 5min to 1 hour depending on the wifi speed

# Then change directory to the new project directory

cd project_name

# To run android emulator (Simple)

press "a" keyboard

# To run android emulator on other cmd tab, run the following command

npx react-native run-android

# To run ios emulator (Simple) [Required MacOsx]

press "i" keyboard

# To run ios emulator on other terminal tab, run the following command [required MacOsx]

npx react-native run-ios

## If in some cases, the emulator is often failed the apps installation due to unknown reason

# Kill all terminal using the following command

press "CTRL + c" in the keyboard

# then enter the following command

npm start -- --reset-cache
